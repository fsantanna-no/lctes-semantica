% Transition arrows.
\makeatletter
\newcommand{\@trans}[2][]{%
  \setbox0=\hbox{$\mathord{\longrightarrow}$}%
  \mathbin{%
    \mathrlap{\longrightarrow}%
    \notblank{#2}{%
      \rlap{\raisebox{-3pt}{\makebox[\wd0][c]{\ensuremath{\scriptstyle#2}}}}%
    }{}%
    \notblank{#1}{%
      \rlap{\raisebox{5pt}{\makebox[\wd0][c]{\ensuremath{\scriptstyle#1}}}}%
    }{}%
    \phantom{\longrightarrow}%
  }%
}
